102 template <typename ValueType>
103 auto set(ValueType&& newValue) -> bool
104 {
105 if (m_value == newValue)
106 return false;
107
108 if constexpr (HasValueChangedEventListener)
109 {
110 auto oldValue = std::move(m_value);
111 m_value = newValue;
112 ValueChangedEvent<T> e{oldValue, m_value};
113 m_valueChangedListeners.notify(e);
114 }
115 else
116 m_value = std::forward<ValueType>(newValue);
117
118 return true;
119 }
120 };
