76 m_window = SDL_CreateWindow(m_parent_window.get_text().c_str(),
77 SDL_WINDOWPOS_CENTERED,
78 SDL_WINDOWPOS_CENTERED,
79 m_parent_window.get_size().get_width(),
80 m_parent_window.get_size().get_height(),
81 SDL_WINDOW_SHOWN | resizable);

84 SDL_Log("Error while creating window: %s", SDL_GetError());

147 void run_event_loop()
148 {
149 m_parent_window().do_layout();
150 m_parent_window().invalidate();
151
152 graphics_sdl2_impl graphics{*m_renderer, m_parent_window.get_size()};
153
154 SDL_Event event;
155 while (SDL_WaitEvent(&event))
156 {
157 switch (event.type)
158 {
159 case SDL_QUIT: return;
160
161 case SDL_WINDOWEVENT:
162 if (event.window.event == SDL_WINDOWEVENT_RESIZED)
163 {
164 const draw::dimension newSize{event.window.data1, event.window.data2};
165
166 resizeGraphics(graphics, newSize);
167 process_control_resized_event(newSize);
168 repaintAll(graphics);
169 }
170
171 break;
172
173 case SDL_MOUSEBUTTONUP:
174 case SDL_MOUSEBUTTONDOWN:
175 trigger_mouse_click_event(event, event.type);
176 break;
177
178 case SDL_MOUSEMOTION:
179 trigger_mouse_moved_event(event);
180 break;
181
182 default:
183 break;
184 }
185
186 const draw::region rgn{{0, 0}, m_parent_window.get_actual_size()};
187 m_parent_window().paint(graphics, rgn);
188 graphics.apply();
189 }
190 }

192 void trigger_mouse_click_event(SDL_Event& e, Uint32 type)
193 {
194 const auto status =
195 type == SDL_MOUSEBUTTONDOWN
196 ? clsn::ui::events::mouse_click_status::pressed
197 : clsn::ui::events::mouse_click_status::released;
198
199 clsn::ui::events::mouse_click_event mouseClickEvent{
200 status, draw::point{e.button.x, e.button.y}};
201
202 m_parent_window.process_mouse_click_event(mouseClickEvent);
203 }
204
205 void trigger_mouse_moved_event(SDL_Event& e)
206 {
207 events::mouse_moved_event _mouse_moved_event{{e.motion.x, e.motion.y}};
208 m_parent_window.process_mouse_moved_event(_mouse_moved_event);
209 }
210
211 void process_control_resized_event(const draw::dimension& newSize)
212 {
213 m_parent_window.set_actual_size(newSize);
214 }
