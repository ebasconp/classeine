86 template <bool B = HasValueChangedEventListener,
87 std::enable_if_t<B, int> = 0>
88 auto add_value_changed_listener(
89 event_listener<value_changed_event<T>> listener) -> int
90 {
91 return m_value_changed_listeners.add(std::move(listener));
92 }
93
99 template <typename ValueType>
100 auto set(ValueType&& newValue) -> bool
101 {
102 if (m_value == newValue)
103 return false;
104
105 if constexpr (HasValueChangedEventListener)
106 {
107 auto oldValue = std::move(m_value);
108 m_value = newValue;
109 value_changed_event<T> e{oldValue, m_value};
110 m_value_changed_listeners.notify(e);
111 }
112 else
113 m_value = std::forward<ValueType>(newValue);
114
115 return true;
116 }
117 };
