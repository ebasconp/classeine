68 void do_layout() override
69 {
70 if (this->get_visible_count() == 0)
71 return;
72
73 m_layout.clear();
74
75 for (auto& e : this->get_elements())
76 {
77 if (!e.m_control->is_visible())
78 continue;
79
80 m_layout.add(
81 {e.m_control->get_actual_position(), e.m_control->get_actual_preferred_size()},
82 e.m_constraint);
83 }
84
85 m_layout.layout(this->get_actual_bounds());
86
87 int count = this->get_count();
88 for (int i = 0, j = 0; i < count; i++)
89 {
90 auto& ctrl = (*this)[i];
91 if (!ctrl.is_visible())
92 continue;
93
94 auto& rgn = m_layout.get_element_at(j).m_region;
95
96 ctrl.set_actual_position(rgn.get_position());
97 ctrl.set_actual_size(rgn.get_size());
98
99 j++;
100 }
101 }
