130 void do_layout() override
131 {
132 if (this->get_visible_count() == 0)
133 return;
134
135 m_layout.clear();
136
137 this->iterate_elements([&](control_and_constraint& e)
138 {
139 if (!e.m_control->is_visible())
140 return;
141
142 m_layout.add(
143 {e.m_control->get_actual_position(), e.m_control->get_actual_preferred_size()},
144 e.m_constraint);
145 });
146
147 m_layout.layout(this->get_actual_bounds());
148
149 int count = get_count();
150 for (int i = 0, j = 0; i < count; i++)
151 {
152 auto& ctrl = (*this)[i];
153 if (!ctrl.is_visible())
154 continue;
155
156 auto& rgn = m_layout.get_element_at(j).m_region;
157
158 ctrl.set_actual_position(rgn.get_position());
159 ctrl.set_actual_size(rgn.get_size());
160
161 j++;
162 }
163 }
